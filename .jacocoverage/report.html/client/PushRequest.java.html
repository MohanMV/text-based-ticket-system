public class PushRequest implements Command {
    
    private States s;
    private String user;
    
    public PushRequest(States s, String user){
        this.s = s;        
        this.user = user;
    }
    
    
    @Override
    public void execute() {
        
        s.stateMain();       
        try {
            s.pushLines(user);
        } catch (IOException ex) {
            Logger.getLogger(PushRequest.class.getName()).log(Level.SEVERE, null, ex);
        }
        s.setDraftTagNull();
        s.clearDraftLines();
    }
    
    @Override
    public void undo(){
        s.stateDrafting();
    }
    
}
